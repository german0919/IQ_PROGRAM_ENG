Funciones locales empleadas

function c=linear_react(n,phi2)
dx = 1/(n-1);
k = zeros(n,1);
for i = 1:n
x = mod((i-1)*dx,0.2);
if x >= 0.05 && x <= 0.15
k(i) = 1;
end
end
A = zeros(n);
b = zeros(n,1);
A(1,1) = 1;
b(1,1) = 1;
for i = 2:n-1
A(i,i-1) = 1;
A(i,i) = -2 - dx^2*phi2*k(i);
A(i,i+1) = 1;
end
A(n,n-1) = 1;
A(n,n) = -1;
A = sparse(A);
c = A\b;
end
